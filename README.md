document.addEventListener('DOMContentLoaded', function() {
    const modal = document.getElementById('modal');
    const modalImg = document.getElementById('modal-img');
    const caption = document.querySelector('.caption');
    const closeBtn = document.querySelector('.modal-close');
    const prevBtn = document.querySelector('.prev');
    const nextBtn = document.querySelector('.next');

    let currentIndex = 0;
    let galleryImages = [];

    // Собираем все изображения из галерей
    const galleries = document.querySelectorAll('.image-gallery');
    galleries.forEach(gallery => {
        const images = gallery.querySelectorAll('img');
        galleryImages = [...galleryImages, ...images];
    });

    function openModal(index) {
        currentIndex = index;
        const img = galleryImages[currentIndex];
        modalImg.src = img.src;
        caption.textContent = img.alt;
        modal.style.display = 'block';
        document.body.style.overflow = 'hidden';
    }

    function closeModal() {
        modal.style.display = 'none';
        document.body.style.overflow = 'auto';
    }

    function showPrev() {
        currentIndex = (currentIndex - 1 + galleryImages.length) % galleryImages.length;
        openModal(currentIndex);
    }

    function showNext() {
        currentIndex = (currentIndex + 1) % galleryImages.length;
        openModal(currentIndex);
    }

    // Обработчик для открытия модального окна по клику на изображение
    galleryImages.forEach((img, index) => {
        img.addEventListener('click', function() {
            openModal(index);
        });
    });

    // Обработчики для элементов модального окна
    closeBtn.addEventListener('click', closeModal);
    prevBtn.addEventListener('click', showPrev);
    nextBtn.addEventListener('click', showNext);

    // Закрытие по клику вне изображения и по клавише Esc
    modal.addEventListener('click', function(e) {
        if (e.target === modal) closeModal();
    });
    document.addEventListener('keydown', function(e) {
        if (e.key === 'Escape') closeModal();
        if (e.key === 'ArrowLeft') showPrev();
        if (e.key === 'ArrowRight') showNext();
    });
});

// 1. Цены по категориям (чтобы легко менять)
  const ticketPrices = {
    'adult-ru': 300,
    'adult-foreign': 300,
    'student-child': 150
  };

  // 2. Находим все поля ввода по ID
  const inputs = {
    'adult-ru': document.getElementById('qty-adult-ru'),
    'adult-foreign': document.getElementById('qty-adult-foreign'),
    'student-child': document.getElementById('qty-student-child')
  };

  const totalDisplay = document.getElementById('total-amount');

  // --- Функция пересчёта итоговой суммы ---
  function recalculateTotal() {
    let total = 0;
    for (const type in ticketPrices) {
      const count = parseInt(inputs[type].value, 10) || 0; // если пусто — считаем 0
      total += count * ticketPrices[type];
    }
    totalDisplay.textContent = total + ' ₽';
  }

  // --- 1. Обработка кнопок «+» и «−» ---
  document.querySelectorAll('.qty-btn').forEach(button => {
    button.addEventListener('click', function() {
      const type = this.dataset.type;          // например: adult-ru
      const current = parseInt(inputs[type].value, 10) || 0;
      let newCount = current;

      if (this.classList.contains('qty-plus')) {
        newCount = current + 1;               // ДОБАВИТЬ
      } else if (this.classList.contains('qty-minus')) {
        newCount = Math.max(0, current - 1);   // УБРАТЬ (не ниже 0)
      }

      inputs[type].value = newCount;           // обновляем поле
      recalculateTotal();                      // пересчитываем сумму
    });
  });

  // --- 2. Обработка ручного ввода (ввести своё количество) ---
  Object.values(inputs).forEach(inputField => {
    inputField.addEventListener('input', function() {
      let val = parseInt(this.value, 10);

      // Если ввели не число или отрицательное — ставим 0
      if (isNaN(val) || val < 0) {
        val = 0;
        this.value = '0';
      }
      recalculateTotal();
    });
  });
</script>
